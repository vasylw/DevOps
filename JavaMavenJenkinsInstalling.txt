#Встановлюємо інстанс з лінухою (Добре себе зарекомендував амазонівський Ubuntu 18.04 Server)
#Оновляємо дані з локального репозиторію пакетів
$ sudo apt update
#Інсталимо джаву (в убунті вона є в локальному репозиторію, інакше треба качати з сайту Оракл) 
$ sudo apt install openjdk-8-jdk

#Інсталимо Мавен (він як і джава є в лок. репозиторії Убунти)
$ sudo apt update
$ sudo apt install maven
# Перевіряємо чи успішно (має видати версію)
$ mvn -version

# Додаємо Дебіан-репозиторій для дженкінса: спочатку ключі для обміну з репозиторієм а потім сам репозиторій
$ wget -q -O - https://pkg.jenkins.io/debian/jenkins.io.key | sudo apt-key add -
$ sudo sh -c 'echo deb http://pkg.jenkins.io/debian-stable binary/ > /etc/apt/sources.list.d/jenkins.list'
# Інсталимо сам Дженкінс, оновивши перед тим дані з локального репозиторію
$ sudo apt update
$ sudo apt install jenkins

#Додаємо правило у фаєрвол, що пропускати на порт дженкінса 8080
$ sudo ufw allow 8080

#Пробуємо заходити браузеом на Дженкінс
192.168.0.10:8080
